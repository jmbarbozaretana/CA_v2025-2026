\Pregunta{1.5}{Complete la siguiente tabla con los valores de ganancias de tensión $G_v$  correspondientes a los valores dados en decibeles ${G_v}_ {dB}$:}

\shortsolution{
\textbf{Respuesta:}
}

\solution{
\textbf{Solución:}

Aplicando $10^{G_{v,dB}/20}$ a cada una de las ganancias de tensión se obtiene:
}

\begin{center}
\begin{tabular}{|c|c|}
\hline
\textbf{Ganancia de tensión ($G_v$)} & \textbf{Ganancia de tensión en dB (${G_v}_ {dB}$)} \\[7pt]
\hline
\solution{$10^{0.5/20} = 1.059$}\shortsolution{$10^{0.5/20} = 1.059$}	&	0.5\\[5pt]
\hline
\solution{$10^{0/20} = 1$}\shortsolution{$10^{0/20} = 1$}	&	0\\[5pt]
\hline
\solution{$10^{46/20} = 199.526$}\shortsolution{$10^{46/20} = 199.526$}	&	46\\[5pt]
\hline
\solution{$10^{10.7/20} = 3.427$}\shortsolution{$10^{10.7/20} = 3.427$}	&	10.7\\[5pt]
\hline
\solution{$10^{-8/20} = 0.398$}\shortsolution{$10^{-8/20} = 0.398$}	&	-8\\[5pt]
\hline
\solution{$10^{-14.5/20} = 0.188$}\shortsolution{$10^{-14.5/20} = 0.188$}	&	-14.5\\[5pt]
\hline
\end{tabular}
\end{center}

\solution{\calif{0.25pts cada valor}}
